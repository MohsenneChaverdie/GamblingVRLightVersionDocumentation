354  private IEnumerator GetRequest(string uri, bool debugMode, JsonFilesToLoad parsedClassEnum)
355  {
356  using (var webRequest = UnityWebRequest.Get(uri))
357  {
358  // Request and wait for the desired page
359  yield return webRequest.SendWebRequest();
360 
361  var pages = uri.Split('/');
362  var page = pages.Length - 1;
363 
364  if (webRequest.isNetworkError)
365  {
366  // to show feedback to user on connection error
367  connectionGetErrorEvent.Raise();
368 
369  Debug.LogError(pages[page] + ": Error: " + webRequest.error);
370  }
371  else
372  {
373  if (debugMode)
374  {
375  Debug.Log(pages[page] + ":\nReceived: " + webRequest.downloadHandler.text);
376  }
377 
378  if (webRequest.isDone)
379  {
380  // to count how many Jsons are done
381  // .. it helps us in loading bar value cal in another script
382  receivedJsonDoneCounter.ApplyChange(1);
383  }
384 
385  // parsing the Json into Stack, so we can use
386  // .. this process can happen in one frame, it is fast, there is no need to count into loading bar
387  JsonParsererExecution(webRequest.downloadHandler.text, parsedClassEnum);
388  }
